概述

wxFrame 是 wxWidgets 中的顶级窗口容器，用于创建带菜单栏、状态栏和主内容区域的应用程序窗口。在 Erlang 中通过 wxErlang 模块使用，适合构建图形化桌面应用。

功能说明
主窗口创建作为应用程序的主窗口容器
菜单栏/工具栏支持可附加 wxMenuBar、wxToolBar
子窗口管理容纳 wxPanel 等子控件
事件处理响应关闭、最大化等窗口事件

创建与初始化

%% 创建一个新窗口
Frame = wxFrame:new(Parent, Id, Title, [Options]).

%% 示例：创建主窗口
start() ->
    Wx = wx:new(),  % 初始化wx环境
    Frame = wxFrame:new(Wx, -1, "五子棋", [
        {size, {800, 600}}, 
        {style, ?wxDEFAULT_FRAME_STYLE}
    ]),
    wxFrame:show(Frame).

关键参数：

  • Parent：父窗口（顶级窗口设为 wx:null()）
  • Id：控件ID（-1表示自动生成）
  • Options：
    • size: 初始大小 {Width, Height}
    • pos: 初始位置 {X, Y}
    • style: 窗口样式（如 ?wxMINIMIZE_BOX）

窗口内容管理

%% 设置窗口中心面板
Panel = wxPanel:new(Frame),
wxFrame:setStatusBar(Frame, wxStatusBar:new(Frame)),  % 添加状态栏
wxFrame:setMenuBar(Frame, wxMenuBar:new()).           % 添加菜单栏

%% 添加子控件
Button = wxButton:new(Panel, 100, [{label, "开始"}, {pos, {50, 50}}]),
wxPanel:connect(Panel, command_button_clicked),  % 绑定事件

函数作用示例

事件处理

  • 所有 GUI 操作（包括绘图）必须在 主 GUI 线程 执行。
  • wx_object 是一个特殊的 OTP 行为，它管理了与 wxWidgets 的事件循环交互。
  • 直接绑定（非 callback 形式）可能导致事件在错误的线程处理，引发：
    • 静默失败（无绘图效果）
    • 随机崩溃
    • 资源泄漏
%% 绑定关闭事件
wxEvtHandler:connect(Panel,paint,[{callback, fun handle_event/2}]),
%% 绘制棋盘
handle_event(#wx{obj=Panel,event=#wxPaint{type=paint}}, _Event) ->
	[{board, Board}] = ets:lookup(chess_state, board),
	draw_chess_board(Panel,Board);

wxEvtHandler:connect/2

通用事件处理

绑定方式行为
直接绑定wxEvtHandler:connect(Panel, paint)
- 事件由 wxWidgets 内部处理，可能绕过 Erlang 的消息队列
- 无法保证在正确的 OTP 进程上下文执行
Callback 形式wxEvtHandler:connect(Panel, paint, [{callback, fun handle_paint/2}])
- 事件会被转换到 wx_object 的进程信箱
- 通过 handle_event/2 回调在正确的线程处理
-spec connect(This::wxEvtHandler(), EventType::wxEventType(), [Option]) -> 'ok' when
      Option :: {'id', integer()} | {'lastId', integer()} | {'skip', boolean()} |
		'callback' | {'callback', function()} | {'userData', term()}.
  • 如果指定callback,则事件触发时会自动调用对应的函数,会覆盖默认行为
callback(#wx{},Event)->term()
  • 如果未指定callback,则事件触发时会自动调用handle_event/2,此模式只能在wx_object中使用
handle_event(#wx{},State)->{noreply,State}

示例

-module(chess_gui).
-export([start/0]).

start() ->
    Wx = wx:new(),
    Frame = wxFrame:new(Wx, -1, "五子棋", [
        {size, {600, 600}},
        {style, ?wxDEFAULT_FRAME_STYLE bor ?wxSTAY_ON_TOP}
    ]),
    
    %% 添加菜单栏
    MenuBar = wxMenuBar:new(),
    FileMenu = wxMenu:new(),
    wxMenu:append(FileMenu, 101, "退出"),
    wxMenuBar:append(MenuBar, FileMenu, "文件"),
    wxFrame:setMenuBar(Frame, MenuBar),
    
    %% 主面板
    Panel = wxPanel:new(Frame),
    wxPanel:setBackgroundColour(Panel, {240, 240, 240}),
    
    %% 事件绑定
    wxFrame:connect(Frame, close_window),
    wxMenu:connect(Frame, command_menu_selected),
    
    %% 显示窗口
    wxFrame:show(Frame),
    main_loop(Frame).

main_loop(Frame) ->
    receive
        #wx{id=101, event=#wxCommand{}} ->  % 点击退出菜单
            wxFrame:close(Frame);
        #wx{event=#wxClose{}} ->            % 点击关闭按钮
            wxFrame:destroy(Frame)
    end.

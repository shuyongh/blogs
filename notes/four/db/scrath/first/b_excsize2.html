5.1 B-树删除

步骤 1：从叶节点删除

从叶节点删除键的代码与其他 nodeReplace* 函数类似。

// 从叶节点移除键
func leafDelete(new BNode, old BNode, idx uint16) {
    new.setHeader(BNODE_LEAF, old.nkeys()-1)
    nodeAppendRange(new, old, 0, 0, idx)
    nodeAppendRange(new, old, idx, idx+1, old.nkeys()-(idx+1))
}

步骤 2：递归删除

结构与插入类似。

// 从树中删除键
func treeDelete(tree *BTree, node BNode, key []byte) BNode {
    // 在哪里找到键？
    idx := nodeLookupLE(node, key)

    // 根据节点类型操作
    switch node.btype() {
    case BNODE_LEAF:
        if !bytes.Equal(key, node.getKey(idx)) {
            return BNode{} // 未找到
        }
        // 删除叶节点中的键
        new := BNode{data: make([]byte, BTREE_PAGE_SIZE)}
        leafDelete(new, node, idx)
        return new
    case BNODE_NODE:
        return nodeDelete(tree, node, idx, key)
    default:
        panic("bad node!")
    }
}

步骤 3：处理内部节点

不同之处在于我们需要合并节点而不是拆分节点。一个节点可以合并到其左兄弟或右兄弟之一。nodeReplace* 函数用于更新链接。

// treeDelete() 的一部分
func nodeDelete(tree *BTree, node BNode, idx uint16, key []byte) BNode {
    // 递归到子节点
    kptr := node.getPtr(idx)
    updated := treeDelete(tree, tree.get(kptr), key)
    if len(updated.data) == 0 {
        return BNode{} // 未找到
    }
    tree.del(kptr)

    new := BNode{data: make([]byte, BTREE_PAGE_SIZE)}

    // 检查合并
    mergeDir, sibling := shouldMerge(tree, node, idx, updated)
    switch {
    case mergeDir < 0: // 左
        merged := BNode{data: make([]byte, BTREE_PAGE_SIZE)}
        nodeMerge(merged, sibling, updated)
        tree.del(node.getPtr(idx - 1))
        nodeReplace2Kid(new, node, idx-1, tree.new(merged), merged.getKey(0))
    case mergeDir > 0: // 右
        merged := BNode{data: make([]byte, BTREE_PAGE_SIZE)}
        nodeMerge(merged, updated, sibling)
        tree.del(node.getPtr(idx + 1))
        nodeReplace2Kid(new, node, idx, tree.new(merged), merged.getKey(0))
    case mergeDir == 0:
        if updated.nkeys() == 0 {
            // 删除后子节点为空且没有兄弟节点可合并。
            // 当父节点只有一个子节点时会发生这种情况。
            // 丢弃空子节点并将父节点作为空节点返回。
            assert(node.nkeys() == 1 && idx == 0)
            new.setHeader(BNODE_NODE, 0)
            // 空节点将在到达根之前被消除。
        } else {
            nodeReplaceKidN(tree, new, node, idx, updated)
        }
    }
    return new
}

关于空节点的额外注意：如果一个节点没有兄弟节点，即使其所有键都被删除，它也不能被合并。在这种情况下，我们需要移除空节点，这也会导致其父节点变成空节点，空节点将向上传播，直到最终被合并。

// 将 2 个节点合并为 1 个
func nodeMerge(new BNode, left BNode, right BNode) {
    new.setHeader(left.btype(), left.nkeys()+right.nkeys())
    nodeAppendRange(new, left, 0, 0, left.nkeys())
    nodeAppendRange(new, right, left.nkeys(), 0, right.nkeys())
}

步骤 4：合并条件

合并的条件是：

  1. 节点小于页面的 1/4（这是任意的）。
  2. 有一个兄弟节点，并且合并结果不超过一页。
// 更新的子节点是否应与兄弟节点合并？
func shouldMerge(
    tree *BTree, node BNode,
    idx uint16, updated BNode,
) (int, BNode) {
    if updated.nbytes() > BTREE_PAGE_SIZE/4 {
        return 0, BNode{}
    }

    if idx > 0 {
        sibling := tree.get(node.getPtr(idx - 1))
        merged := sibling.nbytes() + updated.nbytes() - HEADER
        if merged <= BTREE_PAGE_SIZE {
            return -1, sibling
        }
    }

    if idx+1 < node.nkeys() {
        sibling := tree.get(node.getPtr(idx + 1))
        merged := sibling.nbytes() + updated.nbytes() - HEADER
        if merged <= BTREE_PAGE_SIZE {
            return +1, sibling
        }
    }

    return 0, BNode{}
}

删除代码完成。

5.2 根节点

我们需要随着树的增长和收缩跟踪根节点。让我们从删除开始。

这是 B-树删除的最终接口。当以下情况时，树的高度将减少一：

  1. 根节点不是叶节点。
  2. 根节点只有一个子节点。
func (tree *BTree) Delete(key []byte) bool {
    assert(len(key) != 0)
    assert(len(key) <= BTREE_MAX_KEY_SIZE)

    if tree.root == 0 {
        return false
    }

    updated := treeDelete(tree, tree.get(tree.root), key)
    if len(updated.data) == 0 {
        return false // 未找到
    }

    tree.del(tree.root)
    if updated.btype() == BNODE_NODE && updated.nkeys() == 1 {
        // 移除一个层级
        tree.root = updated.getPtr(0)
    } else {
        tree.root = tree.new(updated)
    }
    return true
}

以下是插入的最终接口：

// 接口
func (tree *BTree) Insert(key []byte, val []byte) {
    assert(len(key) != 0)
    assert(len(key) <= BTREE_MAX_KEY_SIZE)
    assert(len(val) <= BTREE_MAX_VAL_SIZE)

    if tree.root == 0 {
        // 创建第一个节点
        root := BNode{data: make([]byte, BTREE_PAGE_SIZE)}
        root.setHeader(BNODE_LEAF, 2)
        // 一个虚拟键，这使树覆盖整个键空间。
        // 因此查找总是可以找到一个包含的节点。
        nodeAppendKV(root, 0, 0, nil, nil)
        nodeAppendKV(root, 1, 0, key, val)
        tree.root = tree.new(root)
        return
    }

    node := tree.get(tree.root)
    tree.del(tree.root)
    node = treeInsert(tree, node, key, val)

    nsplit, splitted := nodeSplit3(node)
    if nsplit > 1 {
        // 根被拆分，添加一个新层级。
        root := BNode{data: make([]byte, BTREE_PAGE_SIZE)}
        root.setHeader(BNODE_NODE, nsplit)
        for i, knode := range splitted[:nsplit] {
            ptr, key := tree.new(knode), knode.getKey(0)
            nodeAppendKV(root, uint16(i), ptr, key, nil)
        }
        tree.root = tree.new(root)
    } else {
        tree.root = tree.new(splitted[0])
    }
}

它做了两件事：

  1. 当旧根拆分为多个节点时，创建一个新的根节点。
  2. 插入第一个键时，创建第一个叶节点作为根。

这里有一个小技巧。当我们创建第一个节点时，我们向树中插入一个空键。空键是排序顺序中最低可能的键，它使查找函数 nodeLookupLE 总是成功，消除了找不到包含输入键的节点的情况。

5.3 测试 B-树

由于我们的数据结构代码是纯粹的数据结构代码（没有 IO），页面分配代码通过 3 个回调隔离。以下是测试我们 B-树的容器代码，它将页面保存在内存哈希映射中，而不持久化到磁盘。在下一章中，我们将实现持久化，而无需修改 B-树代码。

type C struct {
    tree  BTree
    ref   map[string]string
    pages map[uint64]BNode
}

func newC() *C {
    pages := map[uint64]BNode{}
    return &C{
        tree: BTree{
            get: func(ptr uint64) BNode {
                node, ok := pages[ptr]
                assert(ok)
                return node
            },
            new: func(node BNode) uint64 {
                assert(node.nbytes() <= BTREE_PAGE_SIZE)
                key := uint64(uintptr(unsafe.Pointer(&node.data[0])))
                assert(pages[key].data == nil)
                pages[key] = node
                return key
            },
            del: func(ptr uint64) {
                _, ok := pages[ptr]
                assert(ok)
                delete(pages, ptr)
            },
        },
        ref:   map[string]string{},
        pages: pages,
    }
}

我们使用一个引用映射来记录每个 B-树更新，以便稍后可以验证 B-树的正确性。

func (c *C) add(key string, val string) {
    c.tree.Insert([]byte(key), []byte(val))
    c.ref[key] = val
}

func (c *C) del(key string) bool {
    delete(c.ref, key)
    return c.tree.Delete([]byte(key))
}

测试用例留给读者作为练习。

5.4 结束语

这个 B-树实现非常最小化，但最小化对于学习目的是好的。现实世界的实现可能复杂得多，并包含实用的优化。

我们的 B-树实现有一些简单的改进：

  1. 为叶节点和内部节点使用不同的格式。叶节点不需要指针，内部节点不需要值。这节省了一些空间。
  2. 键或值中的一个长度是冗余的——KV 对的长度可以从下一个键的偏移量推断出来。
  3. 节点的第一个键不需要，因为它继承自其父节点的链接。
  4. 添加校验和以检测数据损坏。

构建键值存储的下一步是将我们的 B-树持久化到磁盘，这是下一章的主题。

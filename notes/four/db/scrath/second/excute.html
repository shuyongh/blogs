14.1 引言

要执行一个语句，该语句被翻译成对现有 DB 接口的函数调用。例如，CREATE TABLE 语句被翻译成一个简单的函数调用：

func qlCreateTable(req *QLCreateTable, tx *DBTX) error {
    return tx.TableNew(&req.Def)
}

其他语句没有那么简单，但它们仍然是粘合代码，本身没有太多功能。执行 SELECT 语句更复杂：

func qlSelect(req *QLSelect, tx *DBReader, out []Record) ([]Record, error) {
    // 记录
    records, err := qlScan(&req.QLScan, tx, out)
    if err != nil {
        return nil, err
    }

    // 输出
    for _, irec := range records {
        orec := Record{Cols: req.Names}
        for _, node := range req.Output {
            ctx := QLEvalContex{env: irec}
            qlEval(&ctx, node)
            if ctx.err != nil {
                return nil, ctx.err
            }
            orec.Vals = append(orec.Vals, ctx.out)
        }
        out = append(out, orec)
    }
    return out, nil
}

它做了两件事：

  1. 基于某些条件获取行（qlScan）。
  2. 通过求值表达式将行转换为输出（qlEval）。

这两件事也是其他语句的一部分。我们先看表达式求值。

14.2 表达式求值

有三个地方需要针对一行求值表达式。它们是 SELECT 语句的表达式列表、FILTER 子句中的条件以及 UPDATE 语句的值。

qlEval 函数用于此类任务。为了保持代码简洁，求值结果（标量值或错误）和当前行被放在 QLEvalContex 结构中。

// 用于求值表达式
type QLEvalContex struct {
    env Record // 可选的行的值
    out Value
    err error
}

// 递归求值表达式
func qlEval(ctx *QLEvalContex, node QLNode)

qlEval 函数递归地求值子表达式，然后执行运算符的计算。列名和字面值（整数和字符串）显然容易处理。

func qlEval(ctx *QLEvalContex, node QLNode) {
    if ctx.err != nil {
        return
    }
    switch node.Type {
    // 引用列
    case QL_SYM:
        if v := ctx.env.Get(string(node.Str)); v != nil {
            ctx.out = *v
        } else {
            qlErr(ctx, "unknown column: %s", node.Str)
        }
    // 字面值
    case QL_I64, QL_STR:
        ctx.out = node.Value
    // 更多；省略...
    default:
        panic("not implemented")
    }
}

运算符也容易处理。在求值表达式时我们还进行类型检查。以 QL_NEG 运算符为例：

// 一元操作
case QL_NEG:
    qlEval(ctx, node.Kids[0])
    if ctx.out.Type == TYPE_INT64 {
        ctx.out.I64 = -ctx.out.I64
    } else {
        qlErr(ctx, "QL_NEG type error")
    }

qlErr 函数在 QLEvalContex 中保存错误。

14.3 获取行

获取行是查询需要做的另一件事。

// 执行查询
func qlScan(req *QLScan, tx *DBReader, out []Record) ([]Record, error) {
    sc := Scanner{}
    err := qlScanInit(req, &sc)
    if err != nil {
        return nil, err
    }
    err = tx.Scan(req.Table, &sc)
    if err != nil {
        return nil, err
    }
    return qlScanRun(req, &sc, out)
}

14.3.1 初始化迭代器

首先，我们需要将 INDEX BY 子句翻译成 Record 类型，它被 Scanner 迭代器使用。INDEX BY 子句要么是：

  1. 点查询。
  2. 开放范围（单个比较）。
  3. 封闭范围（两个不同方向的比较）。

对应的例子：

select expr... from table_name index by a = 1;
select expr... from table_name index by a > 1;
select expr... from table_name index by a > 1 and a < 5;

该子句进入 QLScan 的 Key1 和 Key2 字段。

type QLScan struct {
    Table  string
    Key1   QLNode // 第一个范围或点查询
    Key2   QLNode // 第二个范围
    Filter QLNode
    Offset int64
    Limit  int64
}

翻译只是一些类型检查和类型转换。

// 从 `INDEX BY` 子句创建 `Scanner`
func qlScanInit(req *QLScan, sc *Scanner) error {
    if req.Key1.Type == 0 {
        // 没有 `INDEX BY`；通过主键扫描
        sc.Cmp1, sc.Cmp2 = CMP_GE, CMP_LE
        return nil
    }

    var err error
    sc.Key1, sc.Cmp1, err = qlEvalScanKey(req.Key1)
    if err != nil {
        return err
    }
    if req.Key2.Type != 0 {
        sc.Key2, sc.Cmp2, err = qlEvalScanKey(req.Key1)
        if err != nil {
            return err
        }
    }

    if req.Key1.Type == QL_CMP_EQ && req.Key2.Type != 0 {
        return errors.New("bad `INDEX BY`")
    }
    if req.Key1.Type == QL_CMP_EQ {
        sc.Key2 = sc.Key1
        sc.Cmp1, sc.Cmp2 = CMP_GE, CMP_LE
    }
    return nil
}

qlEvalScanKey 函数用于将比较运算符转换为 Record 类型和 CMP_?? 枚举。那里也没有什么有趣的。

14.3.2 迭代行

在迭代行时，我们需要处理 LIMIT 和 FILTER 子句。

// 从 `Scanner` 获取所有行
func qlScanRun(req *QLScan, sc *Scanner, out []Record) ([]Record, error) {
    for i := int64(0); sc.Valid(); i++ {
        // `LIMIT`
        ok := req.Offset <= i && i < req.Limit
        rec := Record{}
        if ok {
            sc.Deref(&rec)
        }

        // `FILTER`
        if ok && req.Filter.Type != 0 {
            ctx := QLEvalContex{env: rec}
            qlEval(&ctx, req.Filter)
            if ctx.err != nil {
                return nil, ctx.err
            }
            if ctx.out.Type != TYPE_INT64 {
                return nil, errors.New("filter is not of boolean type")
            }
            ok = (ctx.out.I64 != 0)
        }

        if ok {
            out = append(out, rec)
        }
        sc.Next()
    }
    return out, nil
}

14.4 执行语句

SELECT 语句的代码已经列出。让我们添加 DELETE 语句，它没有太大不同。

func qlDelete(req *QLDelete, tx *DBTX) (uint64, error) {
    records, err := qlScan(&req.QLScan, &tx.DBReader, nil)
    if err != nil {
        return 0, err
    }

    tdef := getTableDef(&tx.DBReader, req.Table)
    pk := tdef.Cols[:tdef.PKeys]
    for _, rec := range records {
        key := Record{Cols: pk}
        for _, col := range pk {
            key.Vals = append(key.Vals, *rec.Get(col))
        }
        deleted, err := tx.Delete(req.Table, key)
        assert(err == nil && deleted) // 删除现有行
    }
    return uint64(len(records)), nil
}

其他语句在这一点上应该是显而易见的。

14.5 进一步步骤

正如你所看到的，本章没有添加太多内容。我们查询语言的实现主要是为我们已经实现的功能的粘合代码。在达到这一点之前，数据库如何将 SQL 转换为行对你来说可能是个谜。现在你对数据库有了更好的理解，可能还有一些其他方面你想探索。

你可以尝试向我们的数据库添加更多功能，例如连接、分组和聚合，这些在分析查询中很常见。让这些东西工作起来在这一点上应该不难。

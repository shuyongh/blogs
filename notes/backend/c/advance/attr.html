编译器属性

约 823 字大约 3 分钟

2026-01-04

函数属性

constructor/destructor

在main()之前/之后自动执行函数

__attribute__((constructor))
void init_function(void) {
    printf("在main()之前执行\n");
}

__attribute__((destructor))
void cleanup_function(void) {
    printf("在main()之后执行\n");
}

int main() {
    printf("main函数执行\n");
    return 0;
}

noreturn

标记函数不会返回

#include <stdlib.h>

__attribute__((noreturn))
void fatal_error(const char *msg) {
    fprintf(stderr, "致命错误: %s\n", msg);
    exit(EXIT_FAILURE);
    // 编译器知道这里不会返回
}

format(printf, m, n)

检查printf风格的格式字符串

__attribute__((format(printf, 1, 2)))
void log_message(const char *format, ...) {
    va_list args;
    va_start(args, format);
    vprintf(format, args);
    va_end(args);
}

// 使用：编译器会检查格式字符串
log_message("用户 %s 登录，年龄 %d\n", "Alice", 25);  // 正确
// log_message("用户 %s 登录，年龄 %d\n", 123);  // 警告：类型不匹配

deprecated

标记函数已过时

__attribute__((deprecated("请使用 new_function() 代替")))
void old_function(void) {
    printf("这是旧函数\n");
}

void new_function(void) {
    printf("这是新函数\n");
}

int main() {
    old_function();  // 编译警告：函数已过时
    return 0;
}

warn_unused_result

警告未使用的返回值

__attribute__((warn_unused_result))
int allocate_memory(size_t size) {
    void *ptr = malloc(size);
    return ptr != NULL;
}

int main() {
    allocate_memory(100);  // 警告：返回值未使用
    return 0;
}

always_inline/noinline

强制内联/禁止内联

// 强制内联（即使优化关闭）
__attribute__((always_inline))
static inline int max(int a, int b) {
    return a > b ? a : b;
}

// 禁止内联
__attribute__((noinline))
void debug_log(const char *msg) {
    printf("[DEBUG] %s\n", msg);
}

pure

标记纯函数（无副作用，只依赖参数）

__attribute__((pure))
int calculate_sum(int a, int b) {
    return a + b;  // 无副作用
}

// 编译器可以优化多次调用
int result1 = calculate_sum(10, 20);
int result2 = calculate_sum(10, 20);  // 可能被优化为复用result1

const

标记常量函数（纯函数且不访问全局变量）

__attribute__((const))
int square(int x) {
    return x * x;  // 只依赖参数，不访问全局状态
}

weak

定义弱符号，允许被覆盖

// 默认实现（可以被覆盖）
__attribute__((weak))
void default_handler(void) {
    printf("默认处理器\n");
}

// 在其他文件中可以覆盖
void default_handler(void) {
    printf("自定义处理器\n");
}

变量属性

aligned(n)

指定对齐方式

// 64字节对齐（适合SIMD）
__attribute__((aligned(64)))
float vector[16];

// 结构体对齐
struct Packet {
    char type;
    int data;
} __attribute__((aligned(8)));

packed

紧凑排列结构体（取消填充）

// 紧凑结构体（无填充字节）
struct __attribute__((packed)) NetworkPacket {
    uint8_t type;
    uint32_t length;
    uint8_t data[256];
};

printf("大小: %zu\n", sizeof(struct NetworkPacket));  // 1 + 4 + 256 = 261

cleanup(cleanup_function)

变量离开作用域时自动清理

// 自动释放函数
static void auto_free(void *ptr) {
    void **p = (void**)ptr;
    free(*p);
}

void process_file(void) {
    // 离开作用域时自动释放
    char *buffer __attribute__((cleanup(auto_free))) = malloc(1024);
    
    // 使用buffer...
    // 不需要手动free，离开作用域时自动调用auto_free(&buffer)
}

unused

抑制未使用变量的警告

void callback(int data, void *context __attribute__((unused))) {
    // context可能在某些配置下不使用
    printf("数据: %d\n", data);
}

section("section_name")

将变量/函数放入特定段

// 将变量放入自定义段
__attribute__((section(".my_data")))
int custom_data = 42;

// 将函数放入特定段
__attribute__((section(".text.my_functions")))
void my_function(void) {
    // ...
}

类型属性

transparent_union

透明联合体

typedef union __attribute__((transparent_union)) {
    int i;
    float f;
    void *p;
} Value;

void print_value(Value v) {
    // 可以直接传递int/float/void*，不需要包装
}

int main() {
    print_value(42);      // 直接传递int
    print_value(3.14f);   // 直接传递float
    print_value(NULL);    // 直接传递指针
    return 0;
}
